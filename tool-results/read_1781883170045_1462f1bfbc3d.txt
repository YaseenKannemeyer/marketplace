    85→function timeAgo(dateStr: string): string {
    86→  const date = new Date(dateStr)
    87→  const now = new Date()
    88→  const seconds = Math.floor((now.getTime() - date.getTime()) / 1000)
    89→  if (seconds < 60) return 'Just now'
    90→  if (seconds < 3600) return `${Math.floor(seconds / 60)}m ago`
    91→  if (seconds < 86400) return `${Math.floor(seconds / 3600)}h ago`
    92→  if (seconds < 604800) return `${Math.floor(seconds / 86400)}d ago`
    93→  return date.toLocaleDateString('en-ZA', { day: 'numeric', month: 'short' })
    94→}

   173→  const [progress, setProgress] = useState(0)
   174→  const STORY_DURATION = 5000
   175→  const group = viewingStoryGroup
   176→  const story = group?.stories[viewingStoryIndex]
   177→
   178→  // Handle auto-advance with a separate effect
   179→  useEffect(() => {
   180→    if (!story) return
   181→    setProgress(0)
   182→
   183→    const startTime = Date.now()
   184→    const timer = setInterval(() => {
   185→      const elapsed = Date.now() - startTime
   186→      const pct = Math.min((elapsed / STORY_DURATION) * 100, 100)
   187→      setProgress(pct)
   188→
   189→      if (pct >= 100) {
   190→        clearInterval(timer)
   191→        // Auto-advance after duration
   192→        if (!group) return
   193→        if (viewingStoryIndex < group.stories.length - 1) {
   194→          setViewingStoryIndex(viewingStoryIndex + 1)
   195→        } else {
   196→          const idx = storyGroups.findIndex((g) => g.user.id === group.user.id)
   197→          if (idx < storyGroups.length - 1) {
   198→            setViewingStoryGroup(storyGroups[idx + 1])
   199→          } else {
   200→            setViewingStoryGroup(null)
   201→            onClose()
   202→          }
   203→        }
   204→      }
   205→    }, 50)
   206→
   207→    return () => clearInterval(timer)
   208→  }, [story?.id]) // eslint-disable-line react-hooks/exhaustive-deps
   209→
   210→  if (!group || !story) return null
   211→
   212→  return (
   213→    <div className="fixed inset-0 bg-black flex items-center justify-center" style={{ zIndex: 100 }}>
   214→      {/* Close button */}

   223→            <div
   224→              className="h-full bg-white rounded-full transition-all duration-100"

   226→            />
   227→          </div>
   228→        ))}
   229→      </div>
   230→
   231→      {/* Story content */}
   232→      <div className="relative w-full max-w-md h-[90vh] max-h-[800px] rounded-2xl overflow-hidden cursor-pointer select-none" onClick={(e) => { e.stopPropagation(); goNext() }}>

   480→  const checkPasswordStrength = (pw: string) => {
   481→    let score = 0

   484→    if (/[0-9]/.test(pw)) score++
   485→    if (/[^A-Za-z0-9]/.test(pw)) score++

   903→        const newMsg = await res.json()
   904→        setMessages((prev) => [...prev, newMsg])
   905→      }
   906→    } catch { toast.error('Failed to send message') }
   907→  }
   908→
   909→  const isOwner = userId === selectedListing?.sellerId
   910→
   911→  if (!selectedListing) return null
